- Runtime: `VERDICTS.STRONG_SCORE`, `VERDICTS.CAUTION`, `VERDICTS.WEAK_SCORE` are present and `===` to the corresponding `makeVerdict(...)` result.
- Runtime: Zod round-trip — `z.string().refine(isVerdict).parse('STRONG_SCORE')` succeeds; `.parse('BUY')` throws.

   ```ts
   declare const verdictBrand: unique symbol;
   export type Verdict = string & { readonly [verdictBrand]: true };

   const ALLOWED = ['STRONG_SCORE', 'CAUTION', 'WEAK_SCORE'] as const;
   type VerdictLiteral = (typeof ALLOWED)[number];

   export function makeVerdict(value: VerdictLiteral): Verdict {
     if (!(ALLOWED as readonly string[]).includes(value)) {
       throw new Error(`Invalid verdict: ${value}`);
     }
     return value as unknown as Verdict;
   }

   export function isVerdict(v: unknown): v is Verdict {
     return typeof v === 'string' && (ALLOWED as readonly string[]).includes(v);
   }

   export const VERDICTS = {
     STRONG_SCORE: makeVerdict('STRONG_SCORE'),
     CAUTION:      makeVerdict('CAUTION'),
     WEAK_SCORE:   makeVerdict('WEAK_SCORE'),
   } as const;
   ```

   ```bash
   #!/usr/bin/env bash
   # scripts/forbid-verbs.sh — Phase 1 COMP-01 invariant.
   # Reject any branded recommendation vocabulary in source / prompts / config.
   set -euo pipefail

   PATTERN='\b(BUY|SELL|HOLD|recommend|recommended|target price|you should invest|guaranteed return)\b'

   # Exclusions:
   # - dist / .next / node_modules — build output
   # - this script itself
   # - planning docs that LEGITIMATELY discuss the forbidden vocabulary as compliance subject matter
   EXCLUDE=(
     --glob '!**/dist/**'
     --glob '!**/.next/**'
     --glob '!**/node_modules/**'
     --glob '!**/coverage/**'
     --glob '!scripts/forbid-verbs.sh'
     --glob '!**/RESEARCH.md'
     --glob '!**/PITFALLS.md'
     --glob '!**/PROJECT.md'
     --glob '!**/CLAUDE.md'
     --glob '!**/REQUIREMENTS.md'
     --glob '!**/ROADMAP.md'
     --glob '!.planning/**'
   )

   # Scan apps/, packages/, prompts/, and the root scripts dir (excluding this script).
   if ! command -v rg >/dev/null 2>&1; then
     echo "✗ ripgrep (rg) is required for forbid-verbs.sh. Install via 'brew install ripgrep' or your package manager." >&2
     exit 2
   fi

   if rg -wn "${EXCLUDE[@]}" "$PATTERN" apps packages prompts 2>/dev/null; then
     echo "" >&2
     echo "✗ Forbidden compliance verbs found above (COMP-01)." >&2
     echo "  These verbs imply branded investment recommendations (BUY/SELL/HOLD) and are banned" >&2
     echo "  until SEBI Research Analyst registration. Use STRONG_SCORE / CAUTION / WEAK_SCORE instead." >&2
     exit 1
   fi
   echo "✓ No forbidden verbs."
   ```
