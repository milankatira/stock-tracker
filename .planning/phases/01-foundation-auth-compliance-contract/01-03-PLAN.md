- `POST /auth/signup` with valid `{ email, password, name, consent: { consentVersion, granular } }`:
  - Hashes password with `bcrypt.hash(password, 12)` (per cross-phase contract — BCRYPT_ROUNDS=12).
  - Inserts User document `{ email, name, passwordHash, provider: 'local', emailVerified: false }`.
  - Inserts ConsentRecord `{ userId, consentVersion, granular, ip: req.ip, userAgent: req.headers['user-agent'], timestamp: now, supersedesId: null, source: 'signup' }`.
  - Sets `access_token` cookie (HttpOnly, Secure in prod, SameSite=Lax, Path=/) and `refresh_token` cookie (HttpOnly, Secure in prod, SameSite=Strict, Path=/auth/refresh).

- `POST /auth/signup` with `password.length < 12` → 400 with `kind: 'validation'`.
- `POST /auth/signup` with bad email → 400.
- `consent.recordInitial(userId, dto)` creates a record with `supersedesId: null`; calling `consent.findCurrent(userId)` returns that record.
- `consent.supersede(userId, newConsent)` writes a NEW record whose `supersedesId` is the prior record's `_id`; `findCurrent` now returns the new record.
- The whole signup (user insert + consent insert) is atomic — if consent insert fails, the user is NOT persisted. **Use a mongoose `startSession` + `withTransaction`** (works on the replica set from Plan 01 docker-compose / `MongoMemoryReplSet`).

   ```ts
   async recordInitial(userId: Types.ObjectId, dto: ConsentInput, ctx: { ip: string; userAgent: string }, session?: ClientSession) {
     return this.model.create([{
       userId, consentVersion: dto.consentVersion, granular: dto.granular,
       ip: ctx.ip, userAgent: ctx.userAgent, timestamp: new Date(),
       supersedesId: null, source: 'signup',
     }], { session });

   async supersede(userId: Types.ObjectId, dto: ConsentInput, ctx: { ip: string; userAgent: string }) {
     const current = await this.findCurrent(userId);
     if (!current) throw new NotFoundException('No current consent');
     return this.model.create({
       userId, consentVersion: dto.consentVersion, granular: dto.granular,
       ip: ctx.ip, userAgent: ctx.userAgent, timestamp: new Date(),
       supersedesId: current._id, source: 'reconsent',
     });
   }

   async findCurrent(userId: Types.ObjectId) {
     return this.model.findOne({ userId, supersedesId: null }).sort({ timestamp: -1 });
   }
   ```

10. **`apps/api/src/modules/auth/auth.service.ts`** — wire signup using a Mongo transaction:
    ```ts
    import { InjectConnection } from '@nestjs/mongoose';
    import { Connection } from 'mongoose';

    @Injectable()
    export class AuthService {
      constructor(
        @InjectConnection() private readonly db: Connection,
        private readonly users: UsersService,
        private readonly consent: ConsentService,
        private readonly tokens: TokenService,     // landed in Task 2
      ) {}

      async signup(dto: SignupInput, ctx: { ip: string; userAgent: string }) {
        if (await this.users.findByEmail(dto.email)) {
          throw new ConflictException('Email already registered');
        }
        const passwordHash = await bcrypt.hash(dto.password, BCRYPT_ROUNDS);
        let userId: Types.ObjectId | null = null;
        await this.db.transaction(async (session) => {
          const [user] = await this.users.create({
            email: dto.email, name: dto.name, passwordHash,
            provider: 'local', emailVerified: false,
          }, session);
          userId = user._id;
          await this.consent.recordInitial(user._id, dto.consent, ctx, session);
        });
        const tokens = await this.tokens.issueSession(String(userId!));  // Task 2
        const user = await this.users.findByEmail(dto.email);
        return { user: { id: String(user!._id), email: user!.email, name: user!.name }, tokens };
      }

      async validateEmailPassword(email: string, password: string) {
        const user = await this.users.findByEmailWithHash(email);
        if (!user?.passwordHash) return null;
        return (await bcrypt.compare(password, user.passwordHash)) ? user : null;
      }
    }
    ```
