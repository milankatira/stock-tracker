5. **`apps/api/src/common/filters/all-exceptions.filter.ts`** — global filter mapping Nest exceptions to the `ApiError` discriminated union from `@finsight/shared`:
   ```ts
   import { ArgumentsHost, Catch, ExceptionFilter, HttpException, HttpStatus, Logger } from '@nestjs/common';
   import type { Response } from 'express';
   import type { ApiError } from '@finsight/shared';

   @Catch()
   export class AllExceptionsFilter implements ExceptionFilter {
     private readonly logger = new Logger(AllExceptionsFilter.name);

     catch(exception: unknown, host: ArgumentsHost): void {
       const ctx = host.switchToHttp();
       const res = ctx.getResponse<Response>();
       const { status, body } = this.toApiError(exception);
       res.status(status).json({ error: body });
     }

     private toApiError(exception: unknown): { status: number; body: ApiError } {
       if (exception instanceof HttpException) {
         const status = exception.getStatus();
         const resp = exception.getResponse();
         const message = typeof resp === 'string' ? resp : ((resp as { message?: unknown }).message as string ?? exception.message);
         const details = typeof resp === 'object' ? resp : undefined;
         switch (status) {
           case 400: return { status, body: { kind: 'validation', message, details } };
           case 401: return { status, body: { kind: 'unauthorized', message } };
           case 403: return { status, body: { kind: 'forbidden', message } };
           case 404: return { status, body: { kind: 'not_found', message } };
           case 409: return { status, body: { kind: 'conflict', message } };
           case 429: return { status, body: { kind: 'rate_limited', message } };
           default:  return { status, body: { kind: 'server_error', message } };
         }
       }
       // Truly unknown — log full detail server-side, never leak to client.
       this.logger.error('Unhandled exception', exception instanceof Error ? exception.stack : String(exception));
       return { status: HttpStatus.INTERNAL_SERVER_ERROR, body: { kind: 'server_error', message: 'Internal server error' } };
     }
   }
   ```

7. **`apps/api/src/main.ts`** — wire helmet, cookie-parser, ValidationPipe, AllExceptionsFilter, csrf-csrf. **CSRF wiring is added here but exempts `/auth/google/callback` (Pitfall 9) — Plan 03 will register the auth controller; the CSRF exempt list references the path string.**
   ```ts
   import { NestFactory } from '@nestjs/core';
   import { ValidationPipe } from '@nestjs/common';
   import { ConfigService } from '@nestjs/config';
   import cookieParser from 'cookie-parser';
   import helmet from 'helmet';
   import { doubleCsrf } from 'csrf-csrf';
   import { AppModule } from './app.module';
   import { AllExceptionsFilter } from './common/filters/all-exceptions.filter';

   async function bootstrap() {
     const app = await NestFactory.create(AppModule);
     const cfg = app.get(ConfigService);

     app.use(helmet());
     app.use(cookieParser(cfg.getOrThrow<string>('COOKIE_SECRET')));

     // CSRF — double-submit. Exempt /auth/google/callback (Google's redirect, not ours).
     const { doubleCsrfProtection, generateCsrfToken } = doubleCsrf({
       getSecret: () => cfg.getOrThrow<string>('CSRF_SECRET'),
       getSessionIdentifier: (req) => (req as { ip?: string }).ip ?? 'anon',
       cookieName: '__Host-x-csrf',
       cookieOptions: { sameSite: 'lax', secure: cfg.getOrThrow('NODE_ENV') === 'production', path: '/' },
       ignoredMethods: ['GET', 'HEAD', 'OPTIONS'],
     });
     // Expose token-mint endpoint; Plan 03 will use this in tests.
     app.getHttpAdapter().get('/auth/csrf', (req, res) => { res.json({ token: generateCsrfToken(req as any, res as any) }); });
     // Apply protection to non-safe methods, except OAuth callback.
     app.use((req: any, res: any, next: any) => {
       if (req.path === '/auth/google/callback') return next();
       return doubleCsrfProtection(req, res, next);
     });

     app.useGlobalPipes(new ValidationPipe({ whitelist: true, forbidNonWhitelisted: true, transform: true, transformOptions: { enableImplicitConversion: true } }));
     app.useGlobalFilters(new AllExceptionsFilter());

     await app.listen(cfg.getOrThrow<number>('PORT'));
   }
   bootstrap();
   ```
