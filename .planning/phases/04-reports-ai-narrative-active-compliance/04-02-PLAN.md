   ```typescript
   import { Injectable } from '@nestjs/common';

   export const NARRATIVE_BATCH_QUEUE = 'narrative-batch';

   export interface NarrativeBatchJobData { ticker: string; dataVersionHash: string; }

   @Injectable()
   export class NarrativeBatchQueue {
     constructor(@InjectQueue(NARRATIVE_BATCH_QUEUE) private readonly queue: Queue<NarrativeBatchJobData>) {}

     async enqueueForTicker(ticker: string, dataVersionHash: string) {
       return this.queue.add('narrative', { ticker, dataVersionHash }, {
         jobId: `narrative:${ticker}:${dataVersionHash}`,    // idempotency key â€” same hash same ticker = one job
         attempts: 3,
         backoff: { type: 'exponential', delay: 5000 },
         removeOnComplete: { count: 1000 },
         removeOnFail: { count: 5000 },
       });
     }

     async enqueueBatch(items: NarrativeBatchJobData[]) {
       return Promise.all(items.map((it) => this.enqueueForTicker(it.ticker, it.dataVersionHash)));
     }
   }
   ```

   ```typescript
   import { Processor, WorkerHost } from '@nestjs/bullmq';
   import { Logger } from '@nestjs/common';
   import type { Job } from 'bullmq';
   import { AiService } from '../../ai/ai.service';
   import { ScoringInputsService } from '../../scoring/scoring-inputs.service';
   import { ReportsService } from '../../reports/reports.service';
   import { NarrativeAuditFailedError } from '../../ai/ai.service';
   import { buildFallbackNarrative } from '../../ai/fallback-narrative';
   import { NARRATIVE_BATCH_QUEUE, type NarrativeBatchJobData } from './narrative-batch.queue';

   @Processor(NARRATIVE_BATCH_QUEUE, { concurrency: 4 })
   export class NarrativeBatchProcessor extends WorkerHost {
     private readonly logger = new Logger(NarrativeBatchProcessor.name);

     constructor(
       private readonly ai: AiService,
       private readonly reports: ReportsService,
       private readonly scoringInputs: ScoringInputsService,
     ) { super(); }

     async process(job: Job<NarrativeBatchJobData>) {
       const { ticker, dataVersionHash } = job.data;
       const input = await this.scoringInputs.getLatest(ticker);
       if (input.dataVersionHash !== dataVersionHash) {
         this.logger.log({ msg: 'stale-version-skip', ticker, jobHash: dataVersionHash, currentHash: input.dataVersionHash });
         return { skipped: 'stale-version' as const };
       }

       let narrative; let fallbackUsed = false;
       try {
         narrative = await this.ai.narrative(input);
       } catch (e) {
         if (e instanceof NarrativeAuditFailedError) {
           this.logger.warn({ msg: 'narrative-audit-exhausted', ticker, dataVersionHash });
           narrative = { ...buildFallbackNarrative(input.score.value, input.score.verdict), generatedAt: new Date().toISOString(), auditPassed: true as const };
           fallbackUsed = true;
         } else {
           throw e;                                                  // ComplianceViolationException etc. â†’ queue-level retry/fail
         }
       }

       const swot = await this.ai.swot(input);                       // SWOT failure handling parallel to narrative â€” for v1, on audit-fail use empty SWOT { s:[], w:[], o:[], t:[] }
       await this.reports.upsertNarrative(ticker, {
         narrative: { paragraph: narrative.text, citedSources: narrative.citedSources, generatedAt: new Date().toISOString(), auditPassed: true },
         swot:      { ...this.parseSwot(swot), generatedAt: new Date().toISOString(), auditPassed: true },
         dataVersionHash,
       });
       await this.reports.bustCache(ticker);
       return { ticker, ok: true, fallbackUsed };
     }

     private parseSwot(swotOut: AiOutput): { strengths: string[]; weaknesses: string[]; opportunities: string[]; threats: string[]; citedSources: string[] } {
       // swot prompt emits structured JSON; AiService parses + audits already. Re-parse from the swotOut.text field.
       return JSON.parse(swotOut.text);
     }
   }
   ```
