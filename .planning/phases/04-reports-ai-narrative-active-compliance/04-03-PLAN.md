```typescript
export type Verdict = 'STRONG_SCORE' | 'CAUTION' | 'WEAK_SCORE';

export interface Pillars {
  fundamentals: number; valuation: number; technical: number;
  sentiment: number; risk: number; event: number;
}

export interface InsightsBlock {
  volatility: { stddev1y: number };
  profitConsistency: { profitableQuartersPct: number; window: '12Q' };
  eventSensitivity: { avgAbsReturnOnResultDay: number; baseline: number };
  swot: { strengths: string[]; weaknesses: string[]; opportunities: string[]; threats: string[]; citedSources: string[] };
  promoterHoldings: { latestPct: number; deltaPctVsPrevQ: number };
}

export interface Peer { ticker: string; name: string; score: number; sector?: string; }

export interface Narrative {
  paragraph: string;
  citedSources: string[];
  generatedAt: string;
  auditPassed: true;
}

export interface Disclaimers {
  analysis: string;                  // ALWAYS present
  pastPerformance?: string;          // present when returns are shown
}

export interface StockReportDoc {
  ticker: string;
  name: string;
  sector: string;
  asOf: string;                      // ISO timestamp of last EOD recompute
  dataVersionHash: string;
  score: { value: number; verdict: Verdict; pillars: Pillars; weightsVersion: string };
  fundamentals: { pe: number; pb: number; roe: number; roce: number; debtEquity: number; marketCap: number };
  technicals:  { rsi14: number; macdSignal: 'bullish' | 'bearish' | 'neutral'; dma50: number; dma200: number; price: number; beta: number };
  insights: InsightsBlock;
  peers: Peer[];                                                    // exactly 3
  narrative: Narrative | null;                                      // null only on a brand-new ticker before first narrative-batch run
  disclaimers: Disclaimers;
  dataLineage: Array<{ field: string; source: string; stale: boolean }>;
}
```

   ```typescript
   @Injectable()
   export class ReportsService {
     private readonly logger = new Logger(ReportsService.name);
     constructor(
       private readonly cache: CacheService,                          // Phase 1
       @InjectModel(StockReportDoc.name) private readonly model: Model<StockReportDoc>,
       private readonly http: HttpService,                            // for revalidate webhook
     ) {}

     private readonly REDIS_KEY = (t: string) => `report:stock:${t}`;
     private readonly REDIS_TTL = 60 * 60 * 24;                       // 24h; busted on recompute

     async getStock(ticker: string): Promise<StockReportDoc | null> {
       const key = this.REDIS_KEY(ticker);
       const hit = await this.cache.get<StockReportDoc>(key);
       if (hit) return this.withDisclaimers(hit);
       const doc = await this.model.findOne({ ticker }).lean().exec();
       if (!doc) return null;
       await this.cache.set(key, doc, this.REDIS_TTL);
       return this.withDisclaimers(doc as StockReportDoc);
     }

     private withDisclaimers(doc: StockReportDoc): StockReportDoc {
       return {
         ...doc,
         disclaimers: {
           analysis: ANALYSIS_DISCLAIMER,
           pastPerformance: undefined,                                 // stock report has no returns view; Plan 05 fund report adds it
         },
       };
     }

     async upsertNarrative(ticker: string, payload: UpsertNarrativePayload): Promise<void> {
       await this.model.updateOne(
         { ticker },
         { $set: { narrative: payload.narrative, 'insights.swot': payload.swot, dataVersionHash: payload.dataVersionHash, asOf: new Date().toISOString() } },
         { upsert: true },
       );
       await this.bustCache(ticker);
     }

     async bustCache(ticker: string): Promise<void> {
       await this.cache.del(this.REDIS_KEY(ticker));
       try {
         const hmac = createHmac('sha256', process.env.REVALIDATE_HMAC_SECRET!)
           .update(`stock:${ticker}`).digest('hex');
         await firstValueFrom(this.http.post(`${process.env.REVALIDATE_WEBHOOK_URL}/api/internal/revalidate`,
           { tag: `stock:${ticker}` }, { headers: { 'x-revalidate-hmac': hmac } }));
       } catch (e) {
         this.logger.error({ msg: 'revalidate-webhook-failed', ticker, error: (e as Error).message });
         // best-effort; doc is already persisted
       }
     }
   }
   ```

3. Implement `peer-set.service.ts`:
   ```typescript
   @Injectable()
   export class PeerSetService {
     constructor(
       @InjectModel(Instrument.name) private readonly instruments: Model<Instrument>,
       @InjectModel(StockReportDoc.name) private readonly reports: Model<StockReportDoc>,
       private readonly cache: CacheService,
     ) {}

     async getPeers(ticker: string): Promise<Peer[]> {
       const existing = await this.reports.findOne({ ticker }).select('peers').lean().exec();
       if (existing?.peers?.length === 3) return existing.peers;

       const cacheKey = `peers:${ticker}`;
       const hit = await this.cache.get<Peer[]>(cacheKey);
       if (hit) return hit;

       const subject = await this.instruments.findOne({ ticker }).lean().exec();
       if (!subject) return [];

       const sameSector = await this.instruments.find({ sector: subject.sector, ticker: { $ne: ticker } }).lean().exec();
       const ranked = sameSector
         .map((i) => ({ inst: i, dist: Math.abs(Math.log10(i.marketCap) - Math.log10(subject.marketCap)) }))
         .sort((a, b) => a.dist - b.dist)
         .slice(0, 3);

       const peerScores = await this.reports.find({ ticker: { $in: ranked.map((r) => r.inst.ticker) } })
         .select('ticker score.value').lean().exec();
       const scoreMap = new Map(peerScores.map((p) => [p.ticker, p.score?.value ?? 0]));

       const peers: Peer[] = ranked.map((r) => ({
         ticker: r.inst.ticker,
         name: r.inst.name,
         score: scoreMap.get(r.inst.ticker) ?? 0,
         sector: r.inst.sector,
       }));
       await this.cache.set(cacheKey, peers, 60 * 60 * 24);
       return peers;
     }
   }
   ```
