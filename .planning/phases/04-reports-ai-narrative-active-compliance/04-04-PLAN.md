<objective>
Ship the authenticated stock report page (`/stock/[ticker]`) and its full set of reusable presentation components. The page is a Next.js 15 RSC with Suspense streaming so the score gauge + verdict badge paint above-the-fold within ~500ms, while cards, chart, and peers fill in progressively.    import { cookies } from 'next/headers';
   ```typescript
   import type { StockReportDoc } from '@finsight/shared';

   const API_BASE = process.env.INTERNAL_API_BASE ?? process.env.NEXT_PUBLIC_API_BASE!;

   export async function getStockReport(ticker: string): Promise<StockReportDoc | null> {
     const cookieStore = await cookies();
     const jwt = cookieStore.get('finsight_jwt')?.value;
     const res = await fetch(`${API_BASE}/reports/stock/${ticker}`, {
       headers: jwt ? { Authorization: `Bearer ${jwt}` } : {},
       next: { tags: [`stock:${ticker}`], revalidate: 86400 },
     });
     if (res.status === 404) return null;
     if (!res.ok) throw new Error(`Report fetch failed: ${res.status}`);
     return res.json() as Promise<StockReportDoc>;
   }
   ```

   ```tsx
   import type { Verdict } from '@finsight/shared';

   const ZONE: Record<Verdict, { stroke: string; bg: string }> = {
     STRONG_SCORE: { stroke: 'stroke-emerald-400', bg: 'fill-emerald-500/10' },
     CAUTION:      { stroke: 'stroke-amber-400',   bg: 'fill-amber-500/10'   },
     WEAK_SCORE:   { stroke: 'stroke-rose-400',    bg: 'fill-rose-500/10'    },

   export function ScoreGauge({ score, verdict }: { score: number; verdict: Verdict }) {
     const pct = Math.max(0, Math.min(10, score)) / 10;

     const offset = C * (1 - pct * 0.75);   // 3/4 arc
     const z = ZONE[verdict];
     return (

         </svg>
         <div className="absolute inset-0 flex flex-col items-center justify-center">
           <span className="text-5xl font-bold tabular-nums">{score}</span>
           <span className="text-xs text-muted-foreground">out of 10</span>
         </div>
       </div>
     );
   }
   ```

   ```typescript
   import { NextRequest, NextResponse } from 'next/server';
   import { revalidateTag } from 'next/cache';
   import { createHmac, timingSafeEqual } from 'node:crypto';

   export async function POST(req: NextRequest) {
     const body = await req.json() as { tag?: string };
     const tag = body?.tag;
     const headerHmac = req.headers.get('x-revalidate-hmac') ?? '';
     if (!tag) return NextResponse.json({ ok: false }, { status: 400 });

     const secret = process.env.REVALIDATE_HMAC_SECRET;
     if (!secret) return NextResponse.json({ ok: false }, { status: 500 });

     const expected = createHmac('sha256', secret).update(tag).digest('hex');
     const a = Buffer.from(headerHmac, 'hex');
     const b = Buffer.from(expected, 'hex');
     if (a.length !== b.length || !timingSafeEqual(a, b)) {
       return NextResponse.json({ ok: false }, { status: 401 });
     }

     revalidateTag(tag);
     return NextResponse.json({ ok: true, tag });
   }
   ```
