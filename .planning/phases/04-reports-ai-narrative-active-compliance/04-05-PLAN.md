```typescript
}

export interface FundRisk {
  sharpe1y: number;
  stddev1y: number;       // annualised std-dev of returns
  maxDrawdown1y: number;  // negative number, e.g. -0.18 = -18%

  managerTenureYears: number;

export interface HigherScoringPeer extends FundPeer { scoreDelta: number; }

export interface FundNarrative extends Narrative {}

export interface FundReportDoc {
  schemeCode: string;
  name: string;
  category: string;          // e.g. "Large Cap"
  asOf: string;
  dataVersionHash: string;
  score: { value: number; verdict: Verdict; pillars: Record<string, number>; weightsVersion: string };

  risk: FundRisk;

  narrative: FundNarrative | null;

}
```

   ```typescript
   @Injectable()
   export class FundReportsService {
     private readonly logger = new Logger(FundReportsService.name);
     constructor(
       private readonly cache: CacheService,
       @InjectModel(FundReportDoc.name) private readonly model: Model<FundReportDoc>,
       private readonly http: HttpService,
       private readonly fundPeerSet: FundPeerSetService,
     ) {}

     private readonly REDIS_KEY = (s: string) => `report:fund:${s}`;
     private readonly REDIS_TTL = 60 * 60 * 24;

     async getFund(scheme: string): Promise<FundReportDoc | null> {
       const key = this.REDIS_KEY(scheme);
       const hit = await this.cache.get<FundReportDoc>(key);
       if (hit) return this.augment(hit);
       const doc = await this.model.findOne({ schemeCode: scheme }).lean().exec();
       if (!doc) return null;
       await this.cache.set(key, doc, this.REDIS_TTL);
       return this.augment(doc as FundReportDoc);
     }

     private async augment(doc: FundReportDoc): Promise<FundReportDoc> {
       const out: FundReportDoc = {
         ...doc,
         disclaimers: {
           analysis: ANALYSIS_DISCLAIMER,
           pastPerformance: PAST_PERF_DISCLAIMER,    // ALWAYS for funds
         },
       };
       if (doc.score.value < 6) {
         const higher = await this.fundPeerSet.getHigherScoringPeers(doc.schemeCode);
         if (higher.length > 0) out.higherScoringPeers = higher;
       }
       return out;
     }

     async upsertNarrative(scheme: string, payload: UpsertNarrativePayload): Promise<void> {
       await this.model.updateOne(
         { schemeCode: scheme },
         { $set: { narrative: payload.narrative, dataVersionHash: payload.dataVersionHash, asOf: new Date().toISOString() } },
         { upsert: true },
       );
       await this.bustCache(scheme);
     }

     async bustCache(scheme: string): Promise<void> {
       await this.cache.del(this.REDIS_KEY(scheme));
       try {
         const hmac = createHmac('sha256', process.env.REVALIDATE_HMAC_SECRET!).update(`fund:${scheme}`).digest('hex');
         await firstValueFrom(this.http.post(`${process.env.REVALIDATE_WEBHOOK_URL}/api/internal/revalidate`,
           { tag: `fund:${scheme}` }, { headers: { 'x-revalidate-hmac': hmac } }));
       } catch (e) {
         this.logger.error({ msg: 'fund-revalidate-webhook-failed', scheme, error: (e as Error).message });
       }
     }
   }
   ```

   ```typescript
   @Injectable()
   export class FundPeerSetService {
     constructor(
       @InjectModel(FundReportDoc.name) private readonly funds: Model<FundReportDoc>,
       private readonly cache: CacheService,
     ) {}

     async getPeers(schemeCode: string): Promise<FundPeer[]> { /* same-category 3 closest by AUM tier; cached */ }

     async getHigherScoringPeers(schemeCode: string): Promise<HigherScoringPeer[]> {
       const subject = await this.funds.findOne({ schemeCode }).select('category score.value').lean().exec();
       if (!subject) return [];
       const higher = await this.funds.find({
         category: subject.category,
         schemeCode: { $ne: schemeCode },
         'score.value': { $gt: subject.score.value },
       }).sort({ 'score.value': -1 }).limit(3).select('schemeCode name score.value').lean().exec();
       return higher.map((h) => ({ schemeCode: h.schemeCode, name: h.name, score: h.score.value, scoreDelta: h.score.value - subject.score.value }));
     }
   }
   ```

   ```typescript
   @Processor(FUND_NARRATIVE_BATCH_QUEUE, { concurrency: 4 })
   export class FundNarrativeBatchProcessor extends WorkerHost {
     constructor(
       private readonly ai: AiService,
       private readonly fundReports: FundReportsService,
       private readonly scoringInputs: ScoringInputsService,    // assumes Phase 3 exposes getLatestFund(scheme)
     ) { super(); }

     async process(job: Job<FundNarrativeJobData>) {
       const { schemeCode, dataVersionHash } = job.data;
       const input = await this.scoringInputs.getLatestFund(schemeCode);
       if (input.dataVersionHash !== dataVersionHash) return { skipped: 'stale-version' as const };

       let narrative; let fallbackUsed = false;
       try {
         narrative = await this.ai.narrative(input);            // AiService.narrative is asset-agnostic; prompt schema accepts ScoreInput-shape
       } catch (e) {
         if (e instanceof NarrativeAuditFailedError) {
           narrative = { ...buildFallbackNarrative(input.score.value, input.score.verdict, 'fund'), generatedAt: new Date().toISOString(), auditPassed: true as const };
           fallbackUsed = true;
         } else { throw e; }
       }

       // Funds have a "commentary" object (returns context, risk context, holdings note) instead of stocks' SWOT. v1: keep narrative paragraph only; commentary is v1.1.
       await this.fundReports.upsertNarrative(schemeCode, {
         narrative: { paragraph: narrative.text, citedSources: narrative.citedSources, generatedAt: new Date().toISOString(), auditPassed: true },
         swot: { strengths: [], weaknesses: [], opportunities: [], threats: [], citedSources: [], generatedAt: new Date().toISOString(), auditPassed: true }, // empty for funds; field reused for shape symmetry; UI hides empty SWOT
         dataVersionHash,
       });
       await this.fundReports.bustCache(schemeCode);
       return { schemeCode, ok: true, fallbackUsed };
     }
   }
   ```
