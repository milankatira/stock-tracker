<objective>
Build the **pure, deterministic, explainable stock scoring engine** as a zero-I/O TypeScript module under `apps/api/src/scoring/stock/`. Ship `scoreStock(input) -> ScoreResult` composed of six pillars (Fundamentals 35% / Valuation 20% / Technical 20% / Sentiment 10% / Risk 10% / Event 5%), using `decimal.js` ROUND_HALF_UP at 4dp intermediate and 1dp final, with a graceful 5.0 sentiment fallback when news data is absent. Lock determinism with Vitest snapshots over 10 sector-spanning fixtures + `fast-check` property tests (bounds, weighted-sum reconciliation, monotonicity, structural-equality referential transparency), and run the suite on a Node 20 + Node 22 CI matrix.

Purpose: This module is the core IP — every report, watchlist refresh, comparison verdict, and SEO page reads scores written by this code. Every downstream phase depends on its determinism contract. Plan 02 (funds) follows the same pattern; Plan 03 wires it into BullMQ.

Output: A pure scoring module + per-pillar implementation + 10 fixtures + snapshot suite + property suite + breakdown tests + CI matrix workflow.
</objective>

```typescript
// apps/api/src/scoring/types.ts — exported from index.ts

import type Decimal from 'decimal.js';

export type Direction = 'higher' | 'lower';

export enum Verdict {
  STRONG_SCORE = 'STRONG_SCORE',
  CAUTION = 'CAUTION',
  WEAK_SCORE = 'WEAK_SCORE',
}

export interface SubFactorBreakdown {
  name: string;
  rawValue: number | null;        // null = absent, weight redistributed
  normalisedScore: Decimal;       // 0..10, 4dp
  weightWithinPillar: Decimal;    // 0..1 after redistribution
  source: string;                 // e.g. 'fundamentals.roeTtm'
  direction: Direction;
  isFallback: boolean;            // true if absolute-band fallback used
  isAbsent: boolean;              // true if input missing
}

export interface PillarBreakdown {
  pillar: 'fundamentals' | 'valuation' | 'technical' | 'sentiment' | 'risk' | 'event'
        | 'returns' | 'risk-adjusted' | 'consistency' | 'costs' | 'manager' | 'portfolio';
  pillarScore: Decimal;           // 0..10, 4dp
  weight: Decimal;                // 0..1, pillar weight
  weightedContribution: Decimal;  // pillarScore * weight
  subFactors: SubFactorBreakdown[];
  isFallback: boolean;            // true if pillar emitted neutral 5.0 (all sub-factors absent or contract-defined fallback)
  fallbackReason?: string;        // e.g. 'NO_SENTIMENT_DATA_PRE_PHASE_6'
}

export interface ScoreResult {
  score: number;                  // 1dp number, the final 1-10
  verdict: Verdict;
  pillars: PillarBreakdown[];
  inputHash: string;              // sha256 of canonical input (loader-computed; opaque here)
  scoringEngineVersion: string;   // semver of the scoring package
  computedAt: string;             // ISO-8601 — set by the shell when wrapping; pure core leaves empty string
}

export interface PeerCohortValues {
  // Per-sub-factor peer value vectors. Loader builds this snapshot.
  [subFactorSource: string]: number[];
}

export interface ScoreStockInput {
  instrumentId: string;
  asOfDate: string;               // 'YYYY-MM-DD' in IST
  fundamentals: {
    roeTtm: number | null;
    roceTtm: number | null;
    debtToEquity: number | null;
    revenueCagr3y: number | null;
    profitCagr3y: number | null;
    opMarginTtm: number | null;
  };
  shareholding: {
    promoterPct: number | null;
    pledgedPctOfPromoter: number | null;
    pledgedPctTrend90d: number | null;
  };
  valuation: {
    peTtm: number | null;
    pb: number | null;
    peg: number | null;
    evEbitda: number | null;
    divYield: number | null;
  };
  sectorMedians: { pe: number | null };
  technical: {
    price: number;
    sma50: number | null;
    sma200: number | null;
    rsi14: number | null;
    macd: { macd: number; signal: number } | null;
    bollinger: { upper: number; lower: number } | null;
    return1yVsNifty: number | null;
    return3yVsNifty: number | null;
    beta: number | null;          // informational only — weight 0
  };
  sentiment: {
    last30dAggregate: number | null;       // 0..10 already-aggregated score
    analystConsensus: number | null;       // 0..10 or null
  } | null;                                // null pre-Phase 6 → neutral fallback
  risk: {
    volatility1yAnnualised: number | null;
    maxDrawdown1y: number | null;
    earningsConsistencyPct: number | null;
    auditQualifications: number | null;
  };
  event: {
    meanAbsReturnResults5: number | null;
    meanAbsReturnDividends5: number | null;
    meanAbsReturnSectorNews5: number | null;
  };
  peerCohort: PeerCohortValues;
  _inputHash: string;             // sha256 hex; loader-computed
}

export declare function scoreStock(input: ScoreStockInput): ScoreResult;
export declare function toVerdict(score: Decimal): Verdict;
```
