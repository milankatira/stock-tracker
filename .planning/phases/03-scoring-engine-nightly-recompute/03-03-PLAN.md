**Step 1.3 â€” `apps/api/src/jobs/eod-recompute/score-history.bootstrap.ts`:**

**Critical gotcha:** Mongoose's automatic collection creation does NOT honour the `timeseries:` option on first insert in some driver versions â€” the collection ends up as a regular collection. To guarantee TS semantics, explicitly call `db.createCollection('score_history', { timeseries: { ... }, expireAfterSeconds: ... })` BEFORE first write, with a try/catch on "collection already exists" (NamespaceExists, code 48).

```typescript
import { Injectable, OnApplicationBootstrap, Logger } from '@nestjs/common';
import { InjectConnection } from '@nestjs/mongoose';
import { Connection } from 'mongoose';

@Injectable()
export class ScoreHistoryBootstrap implements OnApplicationBootstrap {
  private readonly logger = new Logger(ScoreHistoryBootstrap.name);
  constructor(@InjectConnection() private readonly conn: Connection) {}

  async onApplicationBootstrap(): Promise<void> {
    try {
      await this.conn.db!.createCollection('score_history', {
        timeseries: {
          timeField: 'computedAt',
          metaField: 'instrumentId',
          granularity: 'hours',
        },
        expireAfterSeconds: 60 * 60 * 24 * 365 * 3,
      });
      this.logger.log('Created score_history time-series collection');
    } catch (err: unknown) {
      const e = err as { code?: number; codeName?: string };
      if (e.code === 48 || e.codeName === 'NamespaceExists') {
        this.logger.log('score_history exists â€” skipping create');
        return;
      }
      throw err;
    }
  }
}
```

**Step 1.4 â€” `apps/api/src/jobs/eod-recompute/score-history.repository.ts`:**
```typescript
@Injectable()
export class ScoreHistoryRepository {
  constructor(
    @InjectModel(ScoreHistory.name) private readonly model: Model<ScoreHistoryDocument>,
  ) {}

  // Time-series collections do NOT support updates â€” only inserts.
  // Idempotency is delegated to BullMQ jobId; if a child re-runs, BullMQ rejects duplicate.
  // This method is INSERT-only; double-write protection is via the jobId contract.
  async insert(doc: Omit<ScoreHistory, never>): Promise<void> {
    await this.model.create(doc);
  }

  async findLatest(instrumentId: Types.ObjectId): Promise<ScoreHistoryDocument | null> {
    return this.model
      .findOne({ instrumentId })
      .sort({ computedAt: -1 })
      .lean()
      .exec();
  }

  async findRange(
    instrumentId: Types.ObjectId,
    fromDate: Date,
    toDate: Date,
  ): Promise<ScoreHistoryDocument[]> {
    return this.model
      .find({ instrumentId, computedAt: { $gte: fromDate, $lte: toDate } })
      .sort({ computedAt: 1 })
      .lean()
      .exec();
  }
}
```

**Note on "upsert vs insert":** Mongoose TS docs warn that time-series collections do NOT support updates/upserts pre-MongoDB 7.0, and even on 7.0+ they support only limited update semantics. We MUST use insert-only. The BullMQ jobId on `${instrumentId}:${asOfDate}` is the idempotency guarantee â€” a duplicate job is rejected before the insert runs.

If a job somehow runs twice (e.g. manual admin re-enqueue with a different jobId or after BullMQ jobId TTL expiry), the result is two documents for the same `(instrumentId, asOfDate)`. The "latest" query naturally returns the freshest by `computedAt`. Document this in code comments; surface in SUMMARY.

**Step 2.1 â€” `apps/api/src/jobs/eod-recompute/redis-score-materialiser.ts`:** The Phase 5 +/- indicator depends on the latest+prev pair. Implementation:
```typescript
import { Injectable } from '@nestjs/common';
import { InjectRedis } from '@nestjs-modules/ioredis';  // or whatever Phase 1's CacheModule exposes
import type Redis from 'ioredis';

export interface ScoreSnapshot {
  score: number;
  verdict: 'STRONG_SCORE' | 'CAUTION' | 'WEAK_SCORE';
  asOfDate: string;
  computedAt: string;        // ISO 8601 UTC
  scoringEngineVersion: string;
}

@Injectable()
export class RedisScoreMaterialiser {
  // TTLs intentionally longer than cron interval so a single missed run does not blank the UI.
  private static readonly LATEST_TTL_SECONDS = 36 * 60 * 60; // 36h
  private static readonly PREV_TTL_SECONDS = 7 * 24 * 60 * 60; // 7d
  private static readonly ASOF_TTL_SECONDS = 36 * 60 * 60;     // 36h

  constructor(@InjectRedis() private readonly redis: Redis) {}

  async writeScore(instrumentId: string, snapshot: ScoreSnapshot): Promise<void> {
    const latestKey = `score:latest:${instrumentId}`;
    const prevKey = `score:prev:${instrumentId}`;
    const asofKey = `score:asof:${instrumentId}`;

    // Atomically move latest â†’ prev, set new latest, set asof hint.
    // Using a Lua script ensures all-or-nothing across a brief window.
    const lua = `
      local current = redis.call('GET', KEYS[1])
      if current then
        redis.call('SET', KEYS[2], current, 'EX', ARGV[3])
      end
      redis.call('SET', KEYS[1], ARGV[1], 'EX', ARGV[2])
      redis.call('SET', KEYS[3], ARGV[4], 'EX', ARGV[5])
      return 1
    `;
    await this.redis.eval(
      lua, 3,
      latestKey, prevKey, asofKey,
      JSON.stringify(snapshot),
      RedisScoreMaterialiser.LATEST_TTL_SECONDS.toString(),
      RedisScoreMaterialiser.PREV_TTL_SECONDS.toString(),
      snapshot.asOfDate,
      RedisScoreMaterialiser.ASOF_TTL_SECONDS.toString(),
    );
  }

  async readLatest(instrumentId: string): Promise<ScoreSnapshot | null> {
    const raw = await this.redis.get(`score:latest:${instrumentId}`);
    return raw ? (JSON.parse(raw) as ScoreSnapshot) : null;
  }

  async readPrev(instrumentId: string): Promise<ScoreSnapshot | null> {
    const raw = await this.redis.get(`score:prev:${instrumentId}`);
    return raw ? (JSON.parse(raw) as ScoreSnapshot) : null;
  }
}
```

**TTL rationale:** Every cache key MUST have a TTL (FOUND-05 global rule). `latest` TTL of 36h > 24h cron interval ensures a single missed run does not blank the UI. `prev` TTL of 7d ensures the +/- indicator survives a week of staleness rather than displaying mysteriously absent indicators.

**Idempotency note:** `writeScore` is naturally idempotent on the SAME asOfDate (same SET overwrites). On a NEW asOfDate, prev is moved before latest is overwritten â€” correct. On a duplicate same-day BullMQ retry (rare; jobId guards against this), prev is moved AGAIN to the same value â€” still correct (no data loss). The Lua script is for atomicity within a single call, not for cross-call idempotency.

**Step 2.2 â€” `apps/api/src/jobs/eod-recompute/eod-recompute.producer.ts`:**
```typescript
import { Injectable, OnApplicationBootstrap, Logger } from '@nestjs/common';
import { InjectQueue } from '@nestjs/bullmq';
import { Queue } from 'bullmq';
import { InstrumentMasterService } from '../../instruments/instrument-master.service'; // Phase 2 contract
import { chunked } from '../../shared/chunked';

const EOD_QUEUE = 'eod-recompute';

@Injectable()
export class EodRecomputeProducer implements OnApplicationBootstrap {
  private readonly logger = new Logger(EodRecomputeProducer.name);

  constructor(
    @InjectQueue(EOD_QUEUE) private readonly queue: Queue,
    private readonly instruments: InstrumentMasterService,
  ) {}

  async onApplicationBootstrap(): Promise<void> {
    // BullMQ 5.x: upsertJobScheduler is idempotent on the scheduler key.
    // [ASSUMED A11] cron at 18:00 IST = 12:30 UTC (Asia/Kolkata is UTC+5:30) â€” see note below.
    // bullmq supports tz in the pattern options. Use '0 18 * * *' with tz: 'Asia/Kolkata'.
    await this.queue.upsertJobScheduler(
      'eod-recompute-daily',
      { pattern: '0 18 * * *', tz: 'Asia/Kolkata' },
      {
        name: 'eod-recompute-parent',
        opts: {
          removeOnComplete: 50,
          removeOnFail: 100,
          attempts: 3,
          backoff: { type: 'exponential', delay: 30_000 },
        },
      },
    );
    this.logger.log('Registered eod-recompute-daily scheduler');
  }

  async fanOut(asOfDate: string): Promise<{ enqueued: number; chunks: number }> {
    const universe = await this.instruments.activeUniverse(asOfDate);
    let enqueued = 0;
    let chunks = 0;
    const CHUNK = 100;
    for (const batch of chunked(universe, CHUNK)) {
      await this.queue.addBulk(batch.map((inst) => ({
        name: 'eod-recompute-child',
        data: {
          instrumentId: inst.id.toString(),
          instrumentType: inst.type,
          asOfDate,
        },
        opts: {
          jobId: `${inst.id}:${asOfDate}`,
          attempts: 3,
          backoff: { type: 'exponential', delay: 5_000 },
          removeOnComplete: true,
          removeOnFail: 500,
        },
      })));
      enqueued += batch.length;
      chunks += 1;
    }
    return { enqueued, chunks };
  }
}
```

```typescript
@Processor('eod-recompute', { concurrency: 10 })  // [ASSUMED A12] â€” tune after observation
export class EodRecomputeProcessor extends WorkerHost {
  private readonly logger = new Logger(EodRecomputeProcessor.name);

  constructor(
    private readonly stocksLoader: StocksScoreLoader,
    private readonly fundsLoader: FundsScoreLoader,
    private readonly history: ScoreHistoryRepository,
    private readonly redis: RedisScoreMaterialiser,
    private readonly version: ScoringEngineVersionProvider,
    private readonly producer: EodRecomputeProducer,
  ) { super(); }

  async process(job: Job): Promise<void> {
    if (job.name === 'eod-recompute-parent') {
      const asOfDate = formatInTimeZone(new Date(), 'Asia/Kolkata', 'yyyy-MM-dd');
      const result = await this.producer.fanOut(asOfDate);
      this.logger.log(`Fanned out ${result.enqueued} children in ${result.chunks} chunks for ${asOfDate}`);
      return;
    }

    // child
    const { instrumentId, instrumentType, asOfDate } = job.data as {
      instrumentId: string; instrumentType: 'STOCK' | 'FUND'; asOfDate: string;
    };

    try {
      const result = instrumentType === 'STOCK'
        ? scoreStock(await this.stocksLoader.loadScoreInput(instrumentId, asOfDate))
        : scoreFund(await this.fundsLoader.loadScoreInput(instrumentId, asOfDate));

      const computedAt = new Date();
      // Order matters: Mongo write FIRST (durable), Redis second (cache).
      // If Mongo succeeds and Redis fails, BullMQ retries â€” Mongo insert is idempotent on jobId (duplicate rejected upstream).
      // If Mongo fails, Redis is not touched â€” the +/- indicator stays on yesterday's data; correct.
      await this.history.insert({
        instrumentId: new Types.ObjectId(instrumentId),
        instrumentType,
        asOfDate,
        computedAt,
        score: result.score,
        verdict: result.verdict,
        pillars: result.pillars,
        scoringEngineVersion: this.version.current(),
      });

      await this.redis.writeScore(instrumentId, {
        score: result.score,
        verdict: result.verdict,
        asOfDate,
        computedAt: computedAt.toISOString(),
        scoringEngineVersion: this.version.current(),
      });
    } catch (err: unknown) {
      this.logger.error(`EOD recompute failed for ${instrumentType} ${instrumentId} @ ${asOfDate}`, err);
      throw err;  // BullMQ retries per `attempts: 3` policy
    }
  }
}
```

**Critical ordering:** Mongo write FIRST, Redis SECOND. If Mongo write fails, Redis stays on the prior day's value â€” correct behaviour. If Redis write fails after Mongo succeeds, BullMQ retries; on retry, Mongo would receive a duplicate insert (one more TS doc â€” acceptable; "latest" query handles it) and Redis succeeds.

**The alternative ordering (Redis first) would mean Redis could show today's score even though Mongo says yesterday â€” that breaks the "/api/score" read path which expects them to match.**

```typescript
  @Post('recompute')

    const jobId = `${dto.instrumentId}:${dto.asOfDate}`;

      { instrumentId: dto.instrumentId, instrumentType: dto.instrumentType, asOfDate: dto.asOfDate },

        attempts: 3,
        backoff: { type: 'exponential', delay: 5_000 },

        removeOnFail: 500,
      },
    );
    return { jobId: job.id!.toString(), status: job.id === jobId ? 'enqueued' : 'duplicate' };
```
