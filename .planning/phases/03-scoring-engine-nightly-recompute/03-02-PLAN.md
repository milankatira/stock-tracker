import { Decimal } from './decimal';
import { scoreReturnsPillar } from './pillars/returns';
import { scoreRiskAdjustedPillar } from './pillars/risk-adjusted';
import { scoreConsistencyPillar } from './pillars/consistency';
import { scoreCostsPillar } from './pillars/costs';
import { scoreManagerPillar } from './pillars/manager';
import { scorePortfolioPillar } from './pillars/portfolio';
import { toVerdict, clampAndRoundFinal } from './compose';
import { VERSION } from '../version';
import type { ScoreFundInput, ScoreResult } from '../types';

const WEIGHTS = {
  returns:       new Decimal('0.35'),
  riskAdjusted:  new Decimal('0.25'),
  consistency:   new Decimal('0.15'),
  costs:         new Decimal('0.10'),
  manager:       new Decimal('0.10'),
  portfolio:     new Decimal('0.05'),
} as const;

export function scoreFund(input: ScoreFundInput): ScoreResult {
  // Hard runtime guard for v1 (Assumption A7) — direct/growth only.
  if (input.planType !== 'DIRECT' || input.option !== 'GROWTH') {
    throw new Error(`scoreFund v1 supports DIRECT/GROWTH only — got ${input.planType}/${input.option}`);
  }
  const pillars = [
    scoreReturnsPillar(input, WEIGHTS.returns),
    scoreRiskAdjustedPillar(input, WEIGHTS.riskAdjusted),
    scoreConsistencyPillar(input, WEIGHTS.consistency),
    scoreCostsPillar(input, WEIGHTS.costs),
    scoreManagerPillar(input, WEIGHTS.manager),
    scorePortfolioPillar(input, WEIGHTS.portfolio),
  ];
  const total = pillars.reduce((acc, p) => acc.plus(p.weightedContribution), new Decimal(0));
  const finalScore = clampAndRoundFinal(total);
  return {
