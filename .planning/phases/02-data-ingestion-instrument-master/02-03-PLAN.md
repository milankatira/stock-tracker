    `data-version-hash.service.ts`:
    ```ts
    import { createHash } from 'node:crypto';
    @Injectable()
    export class DataVersionHashService {
      constructor(private readonly instruments: InstrumentRepository) {}
      async bump(instrumentId: string, patch: { lastPriceTs?: Date; lastFundamentalsTs?: Date; lastNewsTs?: Date }): Promise<string> {
        const doc = await this.instruments.findById(instrumentId);
        if (!doc) throw new Error(`instrument not found: ${instrumentId}`);
        const next = {
          lastPriceTs: patch.lastPriceTs ?? doc.lastPriceTs,
          lastFundamentalsTs: patch.lastFundamentalsTs ?? doc.lastFundamentalsTs,
          lastNewsTs: patch.lastNewsTs ?? doc.lastNewsTs,
        };
        const hash = createHash('sha1').update(JSON.stringify({
          id: instrumentId,
          p: next.lastPriceTs?.toISOString() ?? '',
          f: next.lastFundamentalsTs?.toISOString() ?? '',
          n: next.lastNewsTs?.toISOString() ?? '',
        })).digest('hex');
        await this.instruments.updateOne({ _id: instrumentId }, { $set: { ...next, dataVersionHash: hash } });
        return hash;
      }
    }
    ```

          const cur = bars[i].rawClose;
          if (prev > 0) {

            if (ratio < 0.9 || ratio > 1.1) suspicions.push({ bar: bars[i], ratio });

          if (actions.status === 'ok') {

              if (match) {

    interface StaleCacheRecord<T> { value: T; source: string; fetchedAt: number; }
    @Injectable()
    export class StaleCacheService {
      constructor(@InjectRedis() private readonly redis: Redis,
                  @InjectPinoLogger(StaleCacheService.name) private readonly logger: PinoLogger) {}
      async write<T>(key: string, value: T, source: string, ttlSeconds: number): Promise<void> {
        if (!ttlSeconds || ttlSeconds <= 0) throw new Error('TTL required (data/redis-always-ttl)');
        const rec: StaleCacheRecord<T> = { value, source, fetchedAt: Date.now() };
        await this.redis.set(key, JSON.stringify(rec), 'EX', ttlSeconds);
      }
      async read<T>(key: string): Promise<{ value: T; source: string; fetchedAt: Date } | null> {
        const raw = await this.redis.get(key);
        if (!raw) return null;
        const rec = JSON.parse(raw) as StaleCacheRecord<T>;
        return { value: rec.value, source: rec.source, fetchedAt: new Date(rec.fetchedAt) };
      }
      async evictByPattern(pattern: string): Promise<number> {
        let cursor = '0', evicted = 0;
        do {
          const [next, keys] = await this.redis.scan(cursor, 'MATCH', pattern, 'COUNT', 100);
          cursor = next;
          if (keys.length) evicted += await this.redis.del(...keys);
        } while (cursor !== '0');
        return evicted;
      }
    }

        const TTL = 900;  // 15 min, per RESEARCH.md
        // 1) Yahoo
        try {
          const result = await this.quoteBreakerYahoo.fire(yahooSymbol);
          if (result.status === 'ok') {
            await this.staleCache.write(cacheKey, result.data, result.source, TTL);
            await this.bumpHash(yahooSymbol, { lastPriceTs: result.data.asOf });
            return result;
          }
          // result.status === 'err' (validation, not-found, etc.) — do NOT cache, do NOT bump hash
        } catch (err) {
          // breaker open or wrapped function threw — fall through to NSE
          this.logger.warn({ provider: 'yahoo', yahooSymbol, err: (err as Error).message }, 'yahoo_quote_failed_falling_back');
        }
        // 2) NSE
        try {
          const result = await this.quoteBreakerNse.fire(yahooSymbol);
          if (result.status === 'ok') {
            await this.staleCache.write(cacheKey, result.data, result.source, TTL);
            await this.bumpHash(yahooSymbol, { lastPriceTs: result.data.asOf });
            return result;
          }
        } catch (err) {
          this.logger.warn({ provider: 'nse', yahooSymbol, err: (err as Error).message }, 'nse_quote_failed_falling_back');
        }
        // 3) Stale cache
        const stale = await this.staleCache.read<Quote>(cacheKey);
        if (stale) {
          const stalenessSeconds = Math.floor((Date.now() - stale.fetchedAt.getTime()) / 1000);
          this.logger.info({ yahooSymbol, stalenessSeconds }, 'returning_stale_quote');
          return { status: 'stale', data: stale.value, source: stale.source,
                   fetchedAt: stale.fetchedAt, stalenessSeconds };
        }
        return { status: 'err', reason: 'unknown', message: 'all_providers_exhausted', source: 'price-chain' };
      }

      async getDailyHistory(yahooSymbol: string, from: Date, to: Date): Promise<ProviderResult<OHLCVBar[]>> {

      }

    @Injectable()
    export class TickerTaggerService {
      private symbolIndex: Array<{ id: string; pattern: RegExp }> = [];
      private nameIndex: Array<{ id: string; pattern: RegExp }> = [];

      constructor(
        private readonly instruments: InstrumentRepository,
        @InjectPinoLogger(TickerTaggerService.name) private readonly logger: PinoLogger,
      ) {}

      async onModuleInit(): Promise<void> {
        await this.rebuildIndex();
      }

      async rebuildIndex(): Promise<void> {
        const all = await this.instruments.findAllActive();
        this.symbolIndex = all.map(i => ({
          id: i._id.toString(),
          // Word-boundary match on NSE symbol — avoids RELIANCE matching RELIANCE-EQ etc.
          pattern: new RegExp(`\\b${escapeRegex(i.nseSymbol)}\\b`, 'i'),
        }));
        this.nameIndex = all.map(i => ({
          id: i._id.toString(),
          // Match full-word company name; conservative to avoid Adani over-attribution
          pattern: new RegExp(`\\b${escapeRegex(i.name)}\\b`, 'i'),
        }));
      }

      async tag(items: NewsItem[]): Promise<NewsItem[]> {
        return items.map(item => {
          const haystack = `${item.title} ${item.body ?? ''}`;
          const ids = new Set<string>();
          for (const { id, pattern } of this.symbolIndex) if (pattern.test(haystack)) ids.add(id);
          for (const { id, pattern } of this.nameIndex)   if (pattern.test(haystack)) ids.add(id);
          return { ...item, tickersMentioned: [...ids] };
        });
      }
    }
    function escapeRegex(s: string): string { return s.replace(/[.*+?^${}()|[\]\\]/g, '\\$&'); }
