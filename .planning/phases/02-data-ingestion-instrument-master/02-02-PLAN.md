    import axios, { AxiosError } from 'axios';
    @Injectable()
    export class MfapiAdapter implements FundProvider {
      private readonly base = 'https://api.mfapi.in';
      private readonly limiter = new Bottleneck({ maxConcurrent: 5, minTime: 200 });
      constructor(@InjectPinoLogger(MfapiAdapter.name) private readonly logger: PinoLogger) {}

      async getLatestNav(schemeCode: string): Promise<ProviderResult<NavSnapshot>> {
        try {
          const raw = await this.limiter.schedule(() =>
            pTimeout(axios.get(`${this.base}/mf/${schemeCode}/latest`).then(r => r.data),
                     { milliseconds: 6000 }));
          const parsed = MfapiLatestSchema.parse(raw);
          const latest = parsed.data[0];
          return { status: 'ok', source: 'mfapi.in', fetchedAt: new Date(),
                   data: { schemeCode: parsed.meta.scheme_code,
                           nav: latest.nav,
                           date: this.parseDdMmYyyy(latest.date) } };
        } catch (err) {
          return this.toErr(err, 'mfapi.in');
        }
      }
      // ...getNavHistory, listSchemes follow same pattern
      private parseDdMmYyyy(s: string): Date {
        const [dd, mm, yyyy] = s.split('-');
        return DateTime.fromObject({ day: +dd, month: +mm, year: +yyyy }, { zone: 'Asia/Kolkata' }).toJSDate();
      }
      private toErr(err: unknown, source: string): ProviderResult<never> {
        if (err instanceof z.ZodError) {
          this.logger.error({ source, issues: err.issues }, 'mfapi_schema_validation_failed');
          return { status: 'err', reason: 'validation', message: err.message, source };
        }
        if (axios.isAxiosError(err)) {
          if (err.response?.status === 404) return { status: 'err', reason: 'not-found', message: err.message, source };
          if (err.response && err.response.status >= 500) return { status: 'err', reason: 'upstream-5xx', message: err.message, source };
          if (err.response?.status === 429) return { status: 'err', reason: 'rate-limited', message: err.message, source };
          if (err.code === 'ECONNABORTED') return { status: 'err', reason: 'timeout', message: err.message, source };
        }
        throw err;  // let opossum (Plan 03) count unknown failures
      }
    }
    ```

    import Parser from 'rss-parser';
    import { LRUCache } from 'lru-cache';  // install: pnpm add lru-cache
    @Injectable()
    export class RssNewsAdapter implements NewsProvider {
      private readonly parser = new Parser({ timeout: 5000 });
      private readonly limiter = new Bottleneck({ maxConcurrent: 2, minTime: 1000 });
      private readonly seenGuids = new LRUCache<string, true>({ max: 5000, ttl: 1000 * 60 * 60 * 24 * 7 });
      private readonly feeds = [
        { source: 'moneycontrol-markets', url: 'https://www.moneycontrol.com/rss/marketreports.xml' },
        { source: 'moneycontrol-business', url: 'https://www.moneycontrol.com/rss/business.xml' },
        { source: 'et-markets', url: 'https://economictimes.indiatimes.com/markets/rssfeeds/1977021501.cms' },
      ];
      constructor(@InjectPinoLogger(RssNewsAdapter.name) private readonly logger: PinoLogger) {}

      async getRecent(since: Date): Promise<ProviderResult<NewsItem[]>> {
        try {
          const allItems: NewsItem[] = [];
          for (const feed of this.feeds) {
            const result = await this.limiter.schedule(() =>
              pTimeout(this.parser.parseURL(feed.url) as Promise<unknown>, { milliseconds: 5000 })
                .catch((e) => { this.logger.warn({ feed: feed.source, err: (e as Error).message }, 'rss_feed_failed'); return null; }));
            if (!result) continue;
            for (const raw of (result as any).items ?? []) {
              const parsed = RssItemSchema.parse(raw);
              const guid = parsed.guid ?? parsed.link;
              if (this.seenGuids.has(guid)) continue;
              this.seenGuids.set(guid, true);
              const publishedAt = new Date(parsed.isoDate ?? parsed.pubDate ?? Date.now());
              if (publishedAt <= since) continue;
              allItems.push({
                guid, url: parsed.link, title: parsed.title,
                source: feed.source, publishedAt,
                body: parsed.contentSnippet ?? parsed.content,
              });
            }
          }
          // Adapter returns ALL fresh items (publishedAt > since) from configured feeds.
          // Tagging news items to instruments lives in Plan 03 (TickerTaggerService) — the
          // domain-aware layer that knows the instrument master.
          return { status: 'ok', source: 'rss-news', fetchedAt: new Date(), data: allItems };
        } catch (err) {
          if (err instanceof z.ZodError) {
            this.logger.error({ issues: err.issues }, 'rss_schema_validation_failed');
            return { status: 'err', reason: 'validation', message: err.message, source: 'rss-news' };
          }
          throw err;
        }
      }
    }
    ```
